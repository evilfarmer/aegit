Private Sub SetButtonsCaption()
    On Error GoTo 0
    Dim ctl As control
    For Each ctl In Me.Controls
        'Debug.Print ctl.Name
        On Error GoTo NextIteration
        'Debug.Print ctl.Caption
        If Left$(ctl.Caption, 3) = "CHT" Then
            ctl.Caption = Mid$(ctl.Caption, 4, 2)
        End If
        'Debug.Print ctrl.Caption
NextIteration:
    Next
End Sub

Private Sub SetLabelsCaption()
    On Error GoTo 0
    Dim ctl As control
    For Each ctl In Me.Controls
        'Debug.Print ctl.Name
        On Error GoTo NextIteration
        'If ctl.ControlType = acLabel Then Debug.Print ctl.Name, "Label"
        If ctl.ControlType = acLabel And Left$(ctl.Caption, 3) = "LBL" Then
            ctl.Caption = DLookup("ChartTitle", "_tlkpChart", "ChartNum='CHT" & Mid$(ctl.Caption, 4, 2) & "'")
        End If
NextIteration:
    Next
End Sub
